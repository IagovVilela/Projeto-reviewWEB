⚙️ CONFIGURACAO DO BANCO (OPCIONAL):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   1. Criar banco no MySQL:
      CREATE DATABASE reviews_platform;
   
   2. Configurar .env:
      reviews-platform/.env
      
      DB_DATABASE=reviews_platform
      DB_USERNAME=root
      DB_PASSWORD=sua_senha
   
   3. Executar migrations:
      cd reviews-platform
      php artisan migrate
      php artisan db:seed
